#+begin_src C++ :main no
viso::Case checkCase(const char& c)
{
    if (c >= 65 && c <= 90)
        return UPPERCASE;
    else if (c >= 97 && c <= 122)
        return LOWERCASE;
    else
        return NALPHA;
}
